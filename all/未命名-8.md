import matplotlib.pyplot as plt
import networkx as nx

# 定义节点和边
nodes = [
    'NUG node', 'Other nodes of grid', 'Grid line', 'External Tie Line', 
    'Internal Tie line', 'Issue incentive/dispatching order', 
    'Evaluation results/response scheduling', 'Large power grid', 
    'Power balance capacity', 'Tie-line schedule', 
    'Mutual support between partitions', 'Load aggregator', 'Partition 1', 
    'Partition 2', 'Virtual power plant', 'Partition 3', 'Integral energy', 
    'Partition 4', 'Power selling company', 'Distributed generation', 
    'Distributed energy storage', 'Air conditioning', 'Transferable load', 
    'Other energy'
]

edges = [
    ('NUG node', 'Large power grid'), ('Other nodes of grid', 'Large power grid'), 
    ('Grid line', 'External Tie Line'), ('External Tie Line', 'Internal Tie line'), 
    ('Internal Tie line', 'Power balance capacity'), 
    ('Power balance capacity', 'Tie-line schedule'), 
    ('Tie-line schedule', 'Mutual support between partitions'), 
    ('Mutual support between partitions', 'Load aggregator'), 
    ('Load aggregator', 'Partition 1'), ('Partition 1', 'Partition 2'), 
    ('Partition 2', 'Virtual power plant'), ('Virtual power plant', 'Partition 3'), 
    ('Partition 3', 'Integral energy'), ('Integral energy', 'Partition 4'), 
    ('Partition 4', 'Power selling company'), 
    ('Power selling company', 'Distributed generation'), 
    ('Distributed generation', 'Distributed energy storage'), 
    ('Distributed energy storage', 'Air conditioning'), 
    ('Air conditioning', 'Transferable load'), 
    ('Transferable load', 'Other energy')
]

# 创建有向图
G = nx.DiGraph()

# 添加节点和边
G.add_nodes_from(nodes)
G.add_edges_from(edges)

# 绘制网络图
pos = nx.spring_layout(G)  # 使用弹簧布局
nx.draw(G, pos, with_labels=True, node_size=2000, node_color='skyblue', edge_color='gray', linewidths=2, font_size=10)
plt.title('Smart Grid Dispatching Means and Objects')
plt.show()
